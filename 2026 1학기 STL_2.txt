// array<T, N> - 유일하게 크기 고정된 자료구조
// vector<T> - dynamic array
//------------------------------------------------------------------------------------------------------------
#include <iostream>
#include <vector>
#include <algorithm>
#include "save.h"
#include "YString.h"
using namespace std;

extern bool 관찰;

int main() {
	save("메인.cpp");

	vector<int> v{ 1, 2, 3, 4, 5 };

	// [문제] v에서 3을 제거하라.
	// v.size() == 4
	// v.capacity() == 5
	// v.data() 동일
	// v의 원소 {1, 2, 4, 5}

	// std::remove를 사용하면 된다
	remove(v.begin(), v.end(), 3);
	// OCP(Open-Closed Principle)

	for (int i : v) {
		cout << i << ' ';
	}
	cout << endl;
	
	// design pattern 자료구조 -> (iterator) -> 알고리즘
}

#include <fstream>
